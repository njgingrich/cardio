Form Groups

Put form elements inline by using the field-inline class.

<form>
  <div class="field-inline grid">
    <label class="label col-3">Example Input</label>
    <input class="input"/>
  </div>
  <div class="field-inline grid">
    <label class="label col-3">Example Password</label>
    <input type="password" class="input"/>
  </div>
  <div class="field-inline">
    <div class="select">
      <select>
        <option>Option 1</option>
        <option>Option 2</option>
        <option>Option 3</option>
      </select>
    </div>
    <div class="checkbox">
      <input type="checkbox"/>
      <label class="label">Check 1</label>
    </div>
    <div class="radio-buttons">
      <div class="radio">
        <input id="example4-radio-1" type="radio" name="example4-radio"/>
        <label for="example4-radio-1" class="label">Radio 1</label>
      </div>
      <div class="radio">
        <input id="example4-radio-2" type="radio" name="example4-radio"/>
        <label for="example4-radio-2" class="label">Radio 2</label>
      </div>
    </div>
    <input type="password" class="input"/>
    <button class="primary">Submit</button>
  </div>
</form>

Attaching

Attach buttons to an input by using the attached class on a field.

<form>
  <div class="field-inline">
    <label class="label col-2">Username</label>
    <div class="attached">
      <input class="input"/>
      <button>Check Availability</button>
    </div>
  </div>
  <div class="field-inline attached">
    <button class="success">Button</button>
    <input class="input"/>
    <button>Button 2</button>
  </div>
</form>
